form build_date_range .

  " ##(##)##
  loop at s_ordate.

    gr_jdate-sign = s_ordate-sign.
    gr_jdate-option = s_ordate-option.
    gr_jdate-low = s_ordate-low.

    gr_jdate-high = s_ordate-high.
