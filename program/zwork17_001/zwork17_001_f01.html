      CONTINUE." # ## ## ##### ###
    ENDIF.

    " # ### ## ### ##

          lv_fcurr TYPE string,

FORM load_xls_file .

  DATA: lt_excel TYPE STANDARD TABLE OF alsmex_tabline, " ## -> ## ## ###

  CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
    EXPORTING
      filename    = p_file
      i_begin_col = 1

      i_end_col   = 7      " #### ~ ########

  IF sy-subrc <> 0.
    MESSAGE '## ### ## # ### ######.' TYPE 'E'.

  CLEAR: lv_kurst, lv_fcurr, lv_tcurr, lv_gdatu,
         lv_ukurs, lv_ffact, lv_tfact.
